7.1 City Corporation Diabetes Share and Control Status

This section visualizes diabetes patient share across selected thanas of Chattogram City Corporation using percentages calculated against the total City Corporation population.

The choropleth background shows the percentage share of diabetes patients by thana. Miniature pie charts are placed on top of each thana to show the distribution of fasting control status:

  • Controlled
  • Uncontrolled

The spatial boundary file path is loaded from the .env file using the GPKG_PATH environment variable.

def clean_key(x):
    """
    Standardize text for joining:
    lowercase + remove all non-letter characters.
    """
    if pd.isna(x):
        return None

    return re.sub(r"[^a-z]", "", str(x).lower())


def strip_admin_suffix(x):
    """
    Remove common administrative suffixes from place names.
    """
    if pd.isna(x):
        return None

    pattern = r"\s+(Sadar Upazila|Sadar|Upazila|Thana|PS)$"

    return re.sub(pattern, "", str(x), flags=re.IGNORECASE).strip()


def draw_pie_with_percentages(
    ax,
    x,
    y,
    values,
    colors,
    text_colors,
    radius,
    label_fontsize=6,
    edgecolor="white",
    linewidth=0.55,
):
    """
    Draw a miniature pie chart at a map coordinate
    and place percentage labels inside each slice.
    """
    total = sum(values)

    if total <= 0:
        return

    start_angle = 90

    for value, color, text_color in zip(values, colors, text_colors):
        angle = 360 * value / total
        end_angle = start_angle + angle

        wedge = Wedge(
            center=(x, y),
            r=radius,
            theta1=start_angle,
            theta2=end_angle,
            facecolor=color,
            edgecolor=edgecolor,
            linewidth=linewidth,
            zorder=6,
        )

        ax.add_patch(wedge)

        mid_angle = (start_angle + end_angle) / 2
        mid_angle_rad = np.deg2rad(mid_angle)

        label_x = x + radius * 0.58 * np.cos(mid_angle_rad)
        label_y = y + radius * 0.58 * np.sin(mid_angle_rad)

        label_text = f"{value:.1f}%"

        stroke_color = "white" if text_color == "black" else "black"

        ax.text(
            label_x,
            label_y,
            label_text,
            fontsize=label_fontsize,
            color=text_color,
            ha="center",
            va="center",
            weight="bold",
            zorder=9,
            path_effects=[
                path_effects.withStroke(
                    linewidth=1.25,
                    foreground=stroke_color,
                )
            ],
        )

        start_angle = end_angle

    outline = Circle(
        xy=(x, y),
        radius=radius,
        facecolor="none",
        edgecolor="#222222",
        linewidth=0.55,
        zorder=7,
    )

    ax.add_patch(outline)

bd_ctg = bd_adm3.loc[

control_join = control_city.copy()

# ============================
# Join Spatial, Diabetes, and Control Data

map_dat = bd_ctg.merge(

    how="left",
    left_on="join_key_gadm",
    right_on="join_key_diab",

map_dat = map_dat.merge(

)

].copy()

if len(missing_control_thanas) > 0:

    raise ValueError(f"Unmatched control-status thanas detected: {unmatched}")


# ============================

vals = map_dat["percent"].dropna()

if len(vals) == 0:
    raise ValueError("No percentage values available for binning.")

max_percent = vals.max()

    0.25,
    0.40,
    0.55,
    0.70,
    0.85,
    1.00,

    "0.55–0.70%",
    "0.70–0.85%",
    "0.85–1.00%",
    ">1.00%",
]

map_dat["pct_bin"] = pd.cut(

    labels=diabetes_labels,
    right=True,
    include_lowest=True,
)

map_dat["label_name"] = map_dat["name3_fixed"]

map_dat_plot = map_dat.loc[

].copy()

map_dat_plot["pct_bin"] = pd.Categorical(
    map_dat_plot["pct_bin"],
    categories=diabetes_labels,
    ordered=True,
)

# ============================

control_labels = [
    "Controlled",
    "Uncontrolled",
]

control_colors = sns.color_palette(
    [
        "#F0E442",
        "#0072B2",
    ],
    n_colors=len(control_labels),
).as_hex()

control_color_map = dict(zip(control_labels, control_colors))

control_text_color_map = {
    "Controlled": "black",
    "Uncontrolled": "white",
}


# ============================
# Render Map
# ============================

fig = Figure(figsize=(12, 14))
canvas = FigureCanvas(fig)

ax = fig.subplots()

map_dat_plot.plot(
    ax=ax,
    color=map_dat_plot["fill_color"].tolist(),
    edgecolor=outline_col,
    linewidth=outline_lwd,
    zorder=1,
)


# ============================
# Pie Chart Size
# ============================

minx, miny, maxx, maxy = map_dat_plot.total_bounds

map_width = maxx - minx
map_height = maxy - miny

pie_radius = min(map_width, map_height) * 0.040


# ============================
# Add Labels at True Representative Points
# ============================

label_points = {}

for _, row in map_dat_plot.iterrows():
    point = row.geometry.representative_point()
    label_points[row["label_name"]] = point

    ax.text(
        point.x,
        point.y,
        row["label_name"],
        fontsize=8,
        color="#1A1A1A",
        ha="center",
        va="center",
        bbox={
            "facecolor": "white",
            "alpha": 0.80,
            "edgecolor": "none",
            "boxstyle": "round,pad=0.20",
        },
        zorder=10,
    )


# ============================
# Draw Control Status Pie Charts Slightly Above Labels
# ============================

for _, row in map_dat_plot.iterrows():
    if pd.isna(row["controlled"]) or pd.isna(row["not_controlled"]):
        continue

    point = label_points[row["label_name"]]

    pie_x = point.x
    pie_y = point.y + pie_radius * 1.35

    draw_pie_with_percentages(
        ax=ax,
        x=pie_x,
        y=pie_y,
        values=[
            row["controlled"],
            row["not_controlled"],

        colors=[
            control_color_map["Controlled"],
            control_color_map["Uncontrolled"],
        ],
        text_colors=[
            control_text_color_map["Controlled"],
            control_text_color_map["Uncontrolled"],

        radius=pie_radius,
        label_fontsize=6,
    )

control_legend_handles = [

control_legend = ax.legend(

control_legend.get_frame().set_facecolor("white")

# ============================
# Save PDF and Display Figure
# ============================

ax.set_title("")
ax.set_axis_off()
ax.set_aspect("equal")

control_status_map_pdf = os.path.join(

    pad_inches=0.20,
)

display(fig)

    raise ValueError(f"Unmatched control-status thanas detected: {unmatched}")


# ============================
# Classification Setup
# ============================

vals = map_dat["percent"].dropna()

if len(vals) == 0:
    raise ValueError("No percentage values available for binning.")

max_percent = vals.max()

breaks = [
    0.10,
    0.25,
    0.40,
    0.55,
    0.70,
    0.85,
    1.00,
    max_percent,
]

diabetes_labels = [
    "0.10–0.25%",
    "0.25–0.40%",
    "0.40–0.55%",
    "0.55–0.70%",
    "0.70–0.85%",
    "0.85–1.00%",
    ">1.00%",
]


# ============================
# Apply Classification
# ============================

map_dat["pct_bin"] = pd.cut(
    map_dat["percent"],
    bins=breaks,
    labels=diabetes_labels,
    right=True,
    include_lowest=True,
)

map_dat["label_name"] = map_dat["name3_fixed"]

map_dat_plot = map_dat.loc[
    map_dat["percent"].notna()
].copy()

map_dat_plot["pct_bin"] = pd.Categorical(
    map_dat_plot["pct_bin"],
    categories=diabetes_labels,
    ordered=True,
)


# ============================
# Diabetes Choropleth Palette
# ============================

red_band = [
    "#FCBBA1",
    "#FC9272",
    "#FB6A4A",
    "#EF3B2C",
    "#CB181D",
    "#A50F15",
    "#67000D",
]

diabetes_palette = sns.color_palette(
    red_band,
    n_colors=len(diabetes_labels),
).as_hex()

diabetes_color_map = dict(zip(diabetes_labels, diabetes_palette))

map_dat_plot["fill_color"] = map_dat_plot["pct_bin"].map(diabetes_color_map)

outline_col = "#737373"
outline_lwd = 0.75


# ============================
# Control Status Pie Chart Palette
# ============================

# Controlled: yellow
# Uncontrolled: bluish

control_labels = [
    "Controlled",
    "Uncontrolled",
]

control_colors = sns.color_palette(
    [
        "#F0E442",
        "#0072B2",
    ],
    n_colors=len(control_labels),
).as_hex()

control_color_map = dict(zip(control_labels, control_colors))

control_text_color_map = {
    "Controlled": "black",
    "Uncontrolled": "white",
}


# ============================
# Render Map
# ============================

fig = Figure(figsize=(12, 14))
canvas = FigureCanvas(fig)

ax = fig.subplots()

map_dat_plot.plot(
    ax=ax,
    color=map_dat_plot["fill_color"].tolist(),
    edgecolor=outline_col,
    linewidth=outline_lwd,
    zorder=1,
)


# ============================
# Pie Chart Size
# ============================

minx, miny, maxx, maxy = map_dat_plot.total_bounds

map_width = maxx - minx
map_height = maxy - miny

pie_radius = min(map_width, map_height) * 0.040


# ============================
# Add Labels at True Representative Points
# ============================

label_points = {}

for _, row in map_dat_plot.iterrows():
    point = row.geometry.representative_point()
    label_points[row["label_name"]] = point

    ax.text(
        point.x,
        point.y,
        row["label_name"],
        fontsize=8,
        color="#1A1A1A",
        ha="center",
        va="center",
        bbox={
            "facecolor": "white",
            "alpha": 0.80,
            "edgecolor": "none",
            "boxstyle": "round,pad=0.20",
        },
        zorder=10,
    )


# ============================
# Draw Control Status Pie Charts Slightly Above Labels
# ============================

for _, row in map_dat_plot.iterrows():
    if pd.isna(row["controlled"]) or pd.isna(row["not_controlled"]):
        continue

    point = label_points[row["label_name"]]

    pie_x = point.x
    pie_y = point.y + pie_radius * 1.35

    draw_pie_with_percentages(
        ax=ax,
        x=pie_x,
        y=pie_y,
        values=[
            row["controlled"],
            row["not_controlled"],
        ],
        colors=[
            control_color_map["Controlled"],
            control_color_map["Uncontrolled"],
        ],
        text_colors=[
            control_text_color_map["Controlled"],
            control_text_color_map["Uncontrolled"],
        ],
        radius=pie_radius,
        label_fontsize=6,
    )


# ============================
# Diabetes Legend
# ============================

diabetes_legend_handles = [
    Patch(
        facecolor=diabetes_color_map[label],
        edgecolor=outline_col,
        label=label,
    )
    for label in diabetes_labels
]

diabetes_legend = ax.legend(
    handles=diabetes_legend_handles,
    title="Diabetes \nPatient-to-Population\nRatio(%)",
    loc="upper right",
    frameon=True,
    fontsize=7,
    title_fontsize=8,
)

diabetes_legend.get_frame().set_facecolor("white")
diabetes_legend.get_frame().set_alpha(0.95)

ax.add_artist(diabetes_legend)


# ============================
# Control Status Legend
# ============================

control_legend_handles = [
    Patch(
        facecolor=control_color_map["Controlled"],
        edgecolor="#333333",
        label="Controlled",
    ),
    Patch(
        facecolor=control_color_map["Uncontrolled"],
        edgecolor="#333333",
        label="Uncontrolled",
    ),
]

control_legend = ax.legend(
    handles=control_legend_handles,
    title="Control Status",
    loc=(0.6,0.9),
    frameon=True,
    fontsize=8,
    title_fontsize=9,
)

control_legend.get_frame().set_facecolor("white")
control_legend.get_frame().set_alpha(0.95)


# ============================
# Save PDF and Display Figure
# ============================

ax.set_title("")
ax.set_axis_off()
ax.set_aspect("equal")

control_status_map_pdf = os.path.join(
    REPORT_DIR,
    "chattogram_city_diabetes_control_status_map.pdf",
)

fig.savefig(
    control_status_map_pdf,
    format="pdf",
    bbox_inches="tight",
    pad_inches=0.20,
)

display(fig)
